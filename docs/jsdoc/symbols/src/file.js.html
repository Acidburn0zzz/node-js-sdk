281     util.addEvent = function (type, element, handle) {
282         var elements = [];
283         if (element instanceof Array) {
284             elements = element;
285         } else {
286             elements.push(element);
287         }
288         for (i = 0; i < elements.length; i++) {
289             if (elements[i].attachEvent) {
290                 elements[i].attachEvent("on" + type, function (e) {
291                     handle(e)
292                 });
293             } else if (elements[i].addEventListener) {
294                 elements[i].addEventListener(type, handle, false);
295             }
296         }
297     };

300     function receiveToken(event) {
301         if (event.origin.indexOf("hub.loginradius.com") == -1) {
302             return;
303         }
304 
305         sessionStorage.setItem(token, event.data);
306         module.isauthenticated = true;
307         module.onlogin();

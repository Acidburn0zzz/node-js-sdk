  7 /**The Access Token API is used to get the LoginRadius access token after authentication. It will be valid for the specific duration of time specified in the response.
  8 * @name getAccessToken
  9 * @function  
 10 * @public
 11 * @param token {String} A valid session token, which is added to redirect uri of your website after successfully logged in to provider
 12 * @param secret {String} A unique secret key, which is provided by LoginRadius on your Site manage page
 13 * @param handle {CallbackHandler} callback handler, invoke after getting Userprofile from LoginRadius 
 14 */
 15 exports.getAccessToken = function (token, secret, handle) {
 16     httpGetRequest("https://" + apiDomain + "/api/v2/access_token?token=" + token + "&secret=" + secret, function (data) {
 17         handle(data);
 18     });
 19 };

272 function httpGetRequest(url, handle) {
273     request({ uri: url, headers: { 'User-Agent': 'Nodejs-SDK/2.0 (Nodejs ' + process.versions.node + '; ' + process.platform + ';+5.30;)'} }, function (error, res, body) {
274         handle(JSON.parse(body));
275     });
